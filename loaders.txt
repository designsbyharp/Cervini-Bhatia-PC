    entry: './src/index.js',
    output: {
        filename: 'bundle.js',
        path: path.resolve(__dirname, './dist'),
        publicPath: 'dist/' <-- image is in dist folder

    //this block of code below:
    module: {

        { //for each rule
            test: /\.(png|jpg)$/, //files to use for loader

                'file-loader'
            ]
        }
    ]
}
}
